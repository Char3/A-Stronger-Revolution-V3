ig_proletarians = {
	color = { 100 5 5 }
	texture = "gfx/interface/icons/ig_icons/proletarians.dds"
	
	ideologies = {
		ideology_socialist
		ideology_isolationist
		ideology_populist
		ideology_state_atheism
	}
	
	enable = {
		unlocking_technologies = {
			socialism
		}
	}
	
	traits = {
		ig_trait_fervor
		ig_trait_beligerent_indifference
		ig_revolutionary_belligerence
	}	
	
	on_enable = {}
	on_disable = {}
	
	pop_potential = {
		NOR = {
			is_pop_type = aristocrats
			is_pop_type = capitalists
			is_pop_type = shopkeepers
			is_pop_type = bureaucrats
			is_pop_type = officers
		}			
		NOR = {
			pop_employment_building_group = bg_agriculture
			pop_employment_building_group = bg_ranching
			pop_employment_building_group = bg_plantations
		}
	}
	
	pop_weight = {
		value = 0
	
		add = {
			desc = "POP_MACHINISTS"
			if = {
				limit = { 
					AND = {
						is_pop_type = machinists
						standard_of_living <= 10
					}
				}
				value = 125
			}
		}	
		
		add = {
			desc = "POP_RADICAL"
			if = { 
				limit = {
					OR = {
						radical_fraction = {
							value > 0.40
							pop_type = laborers
						}
						radical_fraction = {
							value > 0.40
							pop_type = peasants
						}
						radical_fraction = {
							value > 0.30
							pop_type = clerks
						}
					}
				}
				value = 75
			}
		}
		
		add = {
			desc = "POP_DISCRIMINATED"
			if = {
				limit = { 
					pop_is_discriminated = yes
				}
				value = 25
			}
		}

		add = {
			desc = "POP_LABORERS"
			if = {
				limit = { 
					AND = {
						is_pop_type = laborers
						standard_of_living <= 10
					}
				}
				value = 150
			}
		}	

		add = {
			desc = "POP_PEASANTS"
			if = {
				limit = {
					AND = {
						is_pop_type = peasants
						standard_of_living <= 10
					}
				}
				value = 100
			}
		}
		
		add = {
			desc = "POP_FARMERS"
			if = {
				limit = {
					AND = {
						is_pop_type = farmers
						standard_of_living <= 7
					}
				}
				value = 100
			}
		}
		
		add = {
			desc = "POP_CLERKS"
			if = {
				limit = {
					AND = {
						is_pop_type = clerks
						standard_of_living <= 11
					}
				}
				value = 100
			}
		}
		
		add = {
			desc = "POP_SERVICEMEN"
			if = {
				limit = {
					AND = {
						is_pop_type = servicemen
						standard_of_living <= 8
					}
				}
				value = 50
			}
		}
		
		add = {
			desc = "POP_ACADEMICS"
			if = {
				limit = {
					AND = {
						is_pop_type = academics
						standard_of_living <= 8
					}
				}
				value = 20
			}
		}
		
		add = {
			desc = "POP_CAPITALISTS"
			if = {
				limit = { 
					is_pop_type = capitalists
				}
				value = -250
			}
		}

		multiply = {
			desc = "NO_LABOR_MOVEMENT"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = labor_movement }
					}
				}
				value = 0.5
			}		
		}		

		multiply = {
			desc = "NO_EGALITARIANISM"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = egalitarianism }
					}
				}
				value = 0.5
			}		
		}

		multiply = {
			desc = "decentralized_power"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}		
		}		
	}
	
	monarch_weight = {
		value = 0
	}	

	commander_weight = {
		value = 0.3
	}	
	
	noble_chance = {
		value = 0.1
	}
	
	female_commander_chance = {	
		value = 0.2
	}
	
	female_politician_chance = {
		value = 0.0

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property
				}
				
			}
			add = {
				value = 0.05
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_in_the_workplace
				}
				
			}
			add = {
				value = 0.05
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}
				
			}
			add = {
				value = 0.05
			}
		}
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	commander_leader_chance = {
		value = 0.25
	}	
}
