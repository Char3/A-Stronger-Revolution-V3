FR_gender_template = { # all gender custom commands are based on it
	type = character
	log_loc_errors = no
	text = {
		trigger = {
			this = { is_female = yes }
		}
		localization_key = CustomLoc_FR_female_
	}
	text = {
		localization_key = CustomLoc_FR_male_
		fallback = yes
	}
}

FR_Country_le = {
	type = country
	log_loc_errors = no

	text = {  
		trigger = {
			OR = {
				this = c:FRA # France
				this = c:SWE # Suède
			}
		}
		localization_key = FR_CusLoc_la # la - singular feminine
	}
	text = {
		trigger = {
			OR = {
				this = c:USA # États-Unis d'Amérique
				this = c:NET # Pays-Bas
			}

		}
		localization_key = FR_CusLoc_les # les - plural masculine
	}
	text = {
		trigger = {
			OR = {
				this = c:GER # Allemagne
				this = c:EGY # Égypte
			}
		}
		localization_key = FR_CusLoc_l # l' - elision singular
	}
	text = {
		localization_key = FR_CusLoc_le # le - singular masculine
		fallback = yes
	}
}
