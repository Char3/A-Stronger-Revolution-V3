usa_purchase_alaska = {
	is_shown = {
		exists = c:USA
		exists = c:RUS
		c:USA = ROOT
		is_subject = no
		NOT = { has_global_variable = sale_of_alaska_happened }
		NOT = {
			has_variable = alaska_purchase_var
		}
		c:RUS = {
			owns_entire_state_region = STATE_ALASKA
			is_subject = no
		}
	}

	possible = {
		owns_entire_state_region = STATE_WASHINGTON
		in_default = no
		taking_loans = no
		is_at_war = no
		exists = c:RUS
		c:RUS = {
			relations:root >= relations_threshold:cordial
			is_at_war = no
		}
	}

	when_taken = {
		set_variable = {
			name = alaska_purchase_var
			value = yes
		}
		set_global_variable = {
			name = sale_of_alaska_happened
			value = yes
		}
		trigger_event = {
			id = alaska.4
			days = 0
			popup = yes
		}
	}
}
