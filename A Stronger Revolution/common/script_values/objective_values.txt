#Player objectives values

global_population = {
	value = 0
	
	every_country = {
		if = {
			add = total_population
		}
	}
}

country_and_subjects_population = {
	value = 0
	
	every_country = {
		if = {
			limit = {
				this = root
				is_subject_of = root
			}
			add = total_population
		}
	}
}

country_and_subjects_population_global_share = {
	value = 0
	if = {
		limit = {
			global_population > 0
		}
		value = country_and_subjects_population
		divide = global_population
	}
}
